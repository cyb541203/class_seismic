01. Integration method: Piecewise exact method¶












In [3]:



    


%%file numerical.m

function [D,V,A]=numerical(nt,dt,M,zeta,K,Wn,Wd,F)

D = zeros(nt,1);
V = zeros(nt,1);
A = zeros(nt,1);

ES = exp(-zeta*Wn*dt)*sin(Wd*dt);
EC = exp(-zeta*Wn*dt)*cos(Wd*dt);

for it = 2:nt
     Ust = F(it-1)/K;
    dUst = (F(it)-F(it-1))/K;

    C2 = D(it-1) + (2*zeta/Wn)*(dUst/dt) - Ust; % B
    C1 =(V(it-1) + zeta*Wn*C2 - dUst/dt)/Wd;    % A   
    
    D(it) = C1*ES + C2*EC + (dUst + Ust) - 2*(zeta/Wn)*(dUst/dt);
    V(it) = C1*(Wd*EC - zeta*Wn*ES) - C2*(Wd*ES + zeta*Wn*EC) + dUst/dt;
    A(it) = -2*zeta*Wn*V(it) - Wn^2*D(it);
end

02. Response Spectrum¶














For ζ = 0%, 3%, 5%, 10%¶












In [4]:



    


for zeta = [0.00 0.03 0.05 0.10]

    % Damping ratio
    % zeta = 0.00;
    % zeta = 0.03;
    % zeta = 0.05;
    % zeta = 0.10;

    F = 9.81*load('tohoku_acc.txt');
    dt= 0.01;
    [nt,nd] = size(F);
    %
    % For each time step
    % [M], [C], [K]
    %
    i = 0;
    for Wn=6.28:-0.005:1.256
    %period 1.0 -> Wn 6.28
    %period 5.0 -> Wn 1.256
        i = i+1;
        M = 1;
        K = Wn^2;
        Wd = Wn*sqrt(1-zeta^2); 
        
        [Xd, Xv, Xa]=numerical(nt,dt,M,zeta,K,Wn,Wd,-F);


        W(i)    = Wn;
        T(i)    = 2*pi/Wn;
        resD(i) = max(abs(Xd));
        resV(i) = max(abs(Xv));
        resA(i) = max(abs(Xa));  % Absolute acceleration
    end
    %Save the spectrums
    respSP = [W', T', resD', resV', resA'];
    filename = ['tohoku_spectrum_', num2str(zeta), '.txt'];
    save(filename, 'respSP', '-ascii');
end
